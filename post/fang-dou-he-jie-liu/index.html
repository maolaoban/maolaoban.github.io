防抖和节流

2021-04-08

防抖就是在触发事件后的n时间内只执行一次，如果在n时间内又触发了事件，则会重新计算函数执行时间。

    //防抖
    function debounce(fn,wait) { 
        let task;
        return function (){
            if(task){
                clearTimeout(task);
                console.log('清理定时器');
            }
            let immediately = !task;
            task = setTimeout(() => {
                task = null;
            }, wait);
            if(immediately){
                fn.apply(this,arguments);
            }
        }
    }

节流就是指连续触发事件但是在n时间内只执行一次函数。

    //节流
    function throttle(fn,wait) {
        let timer;
        return function () {
            if(!timer){
                timer = setTimeout(() => {
                    timer = null;
                    fn.apply(this,arguments);
                }, wait);
            }
          }
    }
