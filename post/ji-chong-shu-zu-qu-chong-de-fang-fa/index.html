几种数组去重的方法

· 2021-04-08 · 
    var arr = [1,23,5,6,7,1,4,5,6,'23',4];

        //ES6 Set去重
        console.log([...new Set(arr)]);

        //双层for循环，splice删去重复的
        function unique(arr){
            for(var i = 0; i < arr.length; i++){
                for(var j = i+1; j < arr.length; j++){
                    if(arr[i] === arr[j]){
                        arr.splice(j,1);
                        j--;
                    }
                }
            }
            return arr;
        }
        console.log(unique(arr));

        //利用indexOf去重
        //indexOf() 方法可返回某个指定的字符串值在字符串中首次出现的位置。
        //没有出现返回-1
        function unique(arr){
            var array = [];
            for(var i = 0; i < arr.length; i++){
                if(array.indexOf(arr[i]) === -1){
                    array.push(arr[i])
                }
            }
            return array;
        }
        console.log(unique(arr));

        //使用includes去重
        function unique(arr){
            var array = [];
            for(var i = 0; i < arr.length; i++){
                if(!array.includes(arr[i])){
                    array.push(arr[i])
                }
            }
            return array;
        }
        console.log(unique(arr));

        //使用reduce去重
        //reduce里的function(prev,cur,index,init):
        //prev 表示上一次调用回调时的返回值，或者初始值 init; 必需
        //cur 表示当前正在处理的数组元素； 必需
        //index 表示当前正在处理的数组元素的索引，若提供 init 值，则索引为0，否则索引为1；可选
        //init 表示初始值。可选
        //reduce(fun,initialValue)最后可传递一个初始值initialValue
        function unique(arr){
            return arr.reduce((pre,cur) => pre.includes(cur) ? pre : [...pre,cur],[]);
        }
        console.log(unique(arr));
